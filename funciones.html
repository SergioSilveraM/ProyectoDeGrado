    def ses_optimizer(self, train, val, lambdas, step):
        """
        Optimizer for SES to find the best lambdas (smoothing level) based on RMSE.
        
        Parameters:
        - train: Training data
        - val: Validation data
        - lambdas: List of lambdas values to test
        - step: Number of steps for forecasting
        
        Returns:
        - best_alpha: The best alpha that minimizes RMSE
        - best_rmse: The RMSE for the best alpha
        """
        best_lambda, best_rmse = None, float("inf")

        # Iterate through each lambdas value
        for lambda_ in lambdas:
            ses_model = SimpleExpSmoothing(train).fit(smoothing_level=lambda_, optimized=False)
            y_pred = ses_model.forecast(step)
            
            # Calculate RMSE
            rmse = np.sqrt(mean_squared_error(val, y_pred))

            # Store the best alpha and RMSE
            if rmse < best_rmse:
                best_lambda, best_rmse = lambda_, rmse

        return best_lambda, best_rmse

    def ses_model_tuning(self, train, val, step, title="Model Tuning - Single Exponential Smoothing"):
        """
        Function to tune the SES model and find the best lambda using RMSE.
        
        Parameters:
        - train: Training data
        - val: Validation data
        - step: Number of steps for forecasting
        - title: Title for the plot
        
        Returns:
        - best_lambda: The best lambda that minimizes RMSE
        - best_rmse: The RMSE for the best lambda
        """
        lambdas = np.arange(0.1, 1, 0.01)  # Range of lambda values to test
        best_lambda, best_rmse = self.ses_optimizer(train, val, lambdas, step=step)

        # Combine training and validation sets
        train_val = pd.concat([train, val])

        # Fit final model using the best lambda
        final_model = SimpleExpSmoothing(train_val).fit(smoothing_level=best_lambda, optimized=False)

        # Forecast on validation set
        y_pred = final_model.forecast(step)

        display(f"Best RMSE on validation set: {best_rmse}")

        # Plot the model predictions
        self.plot_model_SES(train, val, y_pred, title)

        return best_lambda, best_rmse

    def des_model_tuning(self, train, val, test, trend='add'):
        DESResult = namedtuple('DESResult', ['model', 'predictions'])
        # Define alpha and beta ranges
        alphas = np.arange(0.01, 1, 0.10)
        betas = np.arange(0.01, 1, 0.10)
        
        # 1. Optimize using train and val, with step as len(val)
        step_val = len(val)
        best_alpha, best_beta, best_mae = self.des_optimizer(train, val, alphas, betas, trend, step_val)
        
        # 2. Combine train and val for the final model
        train_val = pd.concat([train, val])
        
        # 3. Train the final model with the optimized parameters
        final_model = ExponentialSmoothing(train_val, trend=trend).fit(smoothing_level=best_alpha, smoothing_slope=best_beta)
        
        # 4. Forecast on the test set, using len(test) for step
        step_test = len(test)
        y_pred_test = final_model.forecast(step_test)
        
        # 5. Calculate MAE on the test set
        mae_test = mean_absolute_error(test, y_pred_test)
        
        # 6. Plot all sets (train, val, test) and the test predictions in a single graph
        self.plot_model(train, val, test, y_pred_test, title="Double Exponential Smoothing (Train, Val, Test & Predictions)")
        
        # 7. Print the results 
        print(f"Best Alpha: {best_alpha}, Best Beta: {best_beta}, Test MAE: {mae_test}")
        return DESResult(model=final_model, predictions=y_pred_test)

    # Method to optimize the alpha and beta parameters for DES
    def des_optimizer(self, train, val, alphas, betas, trend, step):
        best_alpha, best_beta, best_mae = None, None, float("inf")
        
        for alpha in alphas:
            for beta in betas:
                des_model = ExponentialSmoothing(train, trend=trend).fit(smoothing_level=alpha, smoothing_slope=beta)
                
                # Forecast using step (which is len(val))
                y_pred_val = des_model.forecast(step)
                
                # Calculate MAE for validation set
                mae = mean_absolute_error(val, y_pred_val)
                
                if mae < best_mae:
                    best_alpha, best_beta, best_mae = alpha, beta, mae

        return best_alpha, best_beta, best_mae

    def tes_optimizer(self, train, val, abg, trend, seasonal, seasonal_periods):
            """
            Optimizes alpha, beta, and gamma parameters for Holt-Winters (TES) model.
            
            Parameters:
            - train: Training dataset.
            - val: Validation dataset.
            - abg: List of combinations of alpha, beta, and gamma to try.
            - trend: Type of trend ('add' for additive or 'mul' for multiplicative).
            - seasonal: Type of seasonality ('add' for additive or 'mul' for multiplicative).
            - seasonal_periods: Number of seasonal periods in the data.

            Returns:
            - Best alpha, beta, gamma, and corresponding MAE.
            """
            best_alpha, best_beta, best_gamma, best_mae = None, None, None, float("inf")
            
            # Iterate over each combination of alpha, beta, and gamma
            for comb in abg:
                # Train the Holt-Winters model with the given parameters
                tes_model = ExponentialSmoothing(
                    train, 
                    trend=trend, 
                    seasonal=seasonal, 
                    seasonal_periods=seasonal_periods
                ).fit(
                    smoothing_level=comb[0], 
                    smoothing_slope=comb[1], 
                    smoothing_seasonal=comb[2]
                )
                
                # Forecast on the validation set
                y_pred = tes_model.forecast(len(val))
                
                # Calculate MAE on the validation set
                mae = mean_absolute_error(val, y_pred)
                
                # Update the best parameters if the current MAE is lower
                if mae < best_mae:
                    best_alpha, best_beta, best_gamma, best_mae = comb[0], comb[1], comb[2], mae

            return best_alpha, best_beta, best_gamma, best_mae

    def tes_model_tuning(self, train, val, test, trend, seasonal, seasonal_periods, title="Model Tuning - Triple Exponential Smoothing"):
        """
        Tunes the Holt-Winters (TES) model by optimizing alpha, beta, and gamma, and evaluates on the test set.
        
        Parameters:
        - train: Training dataset.
        - val: Validation dataset.
        - test: Test dataset.
        - trend: Type of trend ('add' for additive or 'mul' for multiplicative).
        - seasonal: Type of seasonality ('add' for additive or 'mul' for multiplicative).
        - seasonal_periods: Number of seasonal periods in the data.
        - title: Title for the plot (optional).

        Returns:
        - A namedtuple with the final model and test predictions.
        """
        # Define ranges for alpha, beta, and gamma
        alphas = betas = gammas = np.arange(0.01, 1.0, 0.5)
        
        # Create all combinations of alpha, beta, and gamma
        abg = list(itertools.product(alphas, betas, gammas))
        
        # Optimize parameters using the training and validation sets
        best_alpha, best_beta, best_gamma, best_mae = self.tes_optimizer(
            train, val, abg=abg, trend=trend, seasonal=seasonal, seasonal_periods=seasonal_periods
        )
        
        # Train the final model with the best parameters on the combined training and validation set
        train_val = pd.concat([train, val])
        final_model = ExponentialSmoothing(train_val, trend=trend, seasonal=seasonal, seasonal_periods=seasonal_periods).fit(
            smoothing_level=best_alpha, 
            smoothing_slope=best_beta, 
            smoothing_seasonal=best_gamma
        )
        
        # Forecast on the test set
        y_pred_test = final_model.forecast(len(test)+ len(test))[-len(test):]
        
        # Define namedtuple to return both model and predictions
        TESResult = namedtuple('TESResult', ['model', 'predictions'])
        
        # Integrate plot_model to visualize results
        self.plot_model(train, val, test, y_pred_test, title)
        
        return TESResult(final_model, y_pred_test)

    def arma_optimizer(self, train, val, p_max, q_max):
        """
        Optimizes p and q parameters for ARMA model using AIC and BIC, evaluated on the validation set.
        
        Parameters:
        - train: Training dataset.
        - val: Validation dataset.
        - p_max: Maximum value for p parameter.
        - q_max: Maximum value for q parameter.

        Returns:
        - Best model (based on AIC), order (p, q), and validation MAE.
        """
        best_aic = np.inf  # Initialize the best AIC to infinity
        best_bic = np.inf  # Initialize the best BIC to infinity
        best_mae = np.inf  # Initialize the best MAE to infinity
        best_order = None  # Store the best (p, q) order
        best_mdl = None  # Store the best model

        # Iterate over the range of p and q values
        for i in range(p_max + 1):
            for j in range(q_max + 1):
                try:
                    # Fit the ARMA model with current p, q combination
                    tmp_mdl = ARIMA(train, order=(i, 0, j)).fit()  # d=0 for ARMA

                    # Forecast on the validation set
                    y_pred_val = tmp_mdl.forecast(steps=len(val))

                    # Calculate the MAE on the validation set
                    tmp_mae = mean_absolute_error(val, y_pred_val)

                    # Update if current model has better AIC and lower validation MAE
                    if tmp_mae < best_mae:
                        best_aic = tmp_mdl.aic
                        best_bic = tmp_mdl.bic
                        best_mae = tmp_mae
                        best_order = (i, 0, j)
                        best_mdl = tmp_mdl

                except Exception as e:
                    continue  # Skip to the next iteration if model fitting fails

        return best_mdl, best_order, best_aic, best_bic, best_mae
    
    def arma_model_tuning(self, train, val, test, p_max=5, q_max=5, title="ARMA Model"):
        """
        Tunes the ARMA model by optimizing p and q based on validation set, and evaluates on the test set.
        
        Parameters:
        - train: Training dataset.
        - val: Validation dataset.
        - test: Test dataset.
        - p_max: Maximum value for p parameter.
        - q_max: Maximum value for q parameter.
        - title: Title for the plot (optional).

        Returns:
        - Best model, predictions, and validation MAE.
        """
        # Optimize parameters using validation set MAE, AIC and BIC
        best_mdl, best_order, best_aic, best_bic, best_mae = self.arma_optimizer(train, val, p_max, q_max)

        # Combine train and val to fit the final model
        train_val = pd.concat([train, val])
        final_model = ARIMA(train_val, order=best_order).fit()

        # Forecast on the test set
        y_pred_test = final_model.forecast(steps=len(test))

        # Define a namedtuple to return the final model and predictions
        ARMAResult = namedtuple('ARMAResult', ['model', 'predictions', 'aic', 'bic', 'mae'])

        # Optional: call plot_model here if you want to visualize the results
        self.plot_model(train, val, test, y_pred_test, title)
        
        return ARMAResult(final_model, y_pred_test, best_aic, best_bic, best_mae)

        - Best SARIMAX model (based on AIC and validation MAE), order (p, d, q), seasonal_order (P, D, Q), and AIC/BIC values.
        """
        best_aic = np.inf  # Initialize the best AIC to infinity
        best_bic = np.inf  # Initialize the best BIC to infinity
        best_mae = np.inf  # Initialize the best MAE to infinity
        best_order = None  # Store the best (p, d, q) order
        best_seasonal_order = None  # Store the best (P, D, Q) seasonal order
        best_mdl = None  # Store the best SARIMAX model

        # Iterate over the range of p, d, q (non-seasonal) and P, D, Q (seasonal) values
        for p in range(p_max + 1):
            for d in d_range:
                for q in range(q_max + 1):
                    for P in P_range:
                        for D in D_range:
                            for Q in Q_range:
                                try:
                                    # Fit the SARIMAX model with current p, d, q and P, D, Q combinations
                                    tmp_mdl = SARIMAX(train, 
                                                      order=(p, d, q), 
                                                      seasonal_order=(P, D, Q, seasonal_periods)).fit(disp=False)

                                    # Forecast on the validation set
                                    y_pred_val = tmp_mdl.forecast(steps=len(val))

                                    # Calculate the MAE on the validation set
                                    tmp_mae = mean_absolute_error(val, y_pred_val)

                                    # Update if current model has better AIC and lower validation MAE
                                    if tmp_mae < best_mae:
                                        best_aic = tmp_mdl.aic
                                        best_bic = tmp_mdl.bic
                                        best_mae = tmp_mae
                                        best_order = (p, d, q)
                                        best_seasonal_order = (P, D, Q)
                                        best_mdl = tmp_mdl

                                except Exception as e:
                                    continue  # Skip to the next iteration if model fitting fails

        return best_mdl, best_order, best_seasonal_order, best_aic, best_bic, best_mae
    
    def sarimax_model_tuning(self, train, val, test, p_max=5, d_range=range(0, 2), q_max=5, P_range=range(0, 2), D_range=range(0, 2), Q_range=range(0, 2), seasonal_periods=120, title="SARIMAX Model"):
        """
        Tunes the SARIMAX model by optimizing p, d, q, P, D, Q based on validation set, and evaluates on the test set.

        Parameters:
        - train: Training dataset.
        - val: Validation dataset.
        - test: Test dataset.
        - p_max: Maximum value for non-seasonal p parameter.
        - d_range: Range of values for non-seasonal d parameter.
        - q_max: Maximum value for non-seasonal q parameter.
        - P_range: Range of values for seasonal P parameter.
        - D_range: Range of values for seasonal D parameter.
        - Q_range: Range of values for seasonal Q parameter.
        - seasonal_periods: Seasonal period (default is 120).
        - title: Title for the plot (optional).

        Returns:
        - Best model, predictions, and AIC/BIC.
        """
        # Optimize parameters using validation set MAE, AIC and BIC
        best_mdl, best_order, best_seasonal_order, best_aic, best_bic, best_mae = self.sarimax_optimizer(train, val, p_max, d_range, q_max, P_range, D_range, Q_range, seasonal_periods)

        # Combine train and val to fit the final model
        train_val = pd.concat([train, val])
        final_model = SARIMAX(train_val, 
                              order=best_order, 
                              seasonal_order=(best_seasonal_order[0], best_seasonal_order[1], best_seasonal_order[2], seasonal_periods)).fit()

        # Forecast on the test set
        y_pred_test = final_model.forecast(steps=len(test))

        # Define a namedtuple to return the final model and predictions
        SARIMAXResult = namedtuple('SARIMAXResult', ['model', 'predictions', 'aic', 'bic', 'mae'])

        # Optional: call plot_model here if you want to visualize the results
        self.plot_model(train, val, test, y_pred_test, title)
        
        return SARIMAXResult(final_model, y_pred_test, best_aic, best_bic, best_mae)
